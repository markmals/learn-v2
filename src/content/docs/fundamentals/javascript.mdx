## DOM Manipulation

JavaScript's main job in web development is manipulating the DOM:

```js
// Finding elements
const button = document.querySelector('#submit-button');
const articles = document.querySelectorAll('.article-card');
const form = document.getElementById('contact-form');

// Changing content
button.textContent = 'Click me!';
button.innerHTML = '<strong>Click me!</strong>';

// Changing styles
button.style.backgroundColor = '#007acc';
button.classList.add('primary');
button.classList.remove('secondary');
button.classList.toggle('active');

// Creating elements
const newArticle = document.createElement('article');
newArticle.className = 'article-card';
newArticle.innerHTML = `
    <h2>New Article</h2>
    <p>This is a dynamically created article.</p>
`;

// Adding to the page
document.querySelector('.articles-container').appendChild(newArticle);

// Removing elements
const oldArticle = document.querySelector('.outdated');
if (oldArticle) {
    oldArticle.remove();
}
```

## Event Handling

Respond to user interactions:

```js
// Click events
button.addEventListener('click', function(event) {
    event.preventDefault();
    console.log('Button clicked!');
    
    // Change button state
    this.disabled = true;
    this.textContent = 'Processing...';
});

// Form events
form.addEventListener('submit', async (event) => {
    event.preventDefault();
    
    const formData = new FormData(event.target);
    const data = Object.fromEntries(formData.entries());
    
    try {
        const response = await fetch('/api/contact', {
            method: 'POST',
            headers: { 'Content-Type': 'application/json' },
            body: JSON.stringify(data)
        });
        
        if (response.ok) {
            showMessage('Thank you! Message sent successfully.');
            form.reset();
        } else {
            throw new Error('Failed to send message');
        }
    } catch (error) {
        showMessage('Sorry, something went wrong. Please try again.', 'error');
    }
});

// Input events
const searchInput = document.querySelector('#search');
searchInput.addEventListener('input', (event) => {
    const query = event.target.value;
    if (query.length > 2) {
        searchArticles(query);
    }
});

// Scroll events (throttled for performance)
let isScrolling = false;
window.addEventListener('scroll', () => {
    if (!isScrolling) {
        requestAnimationFrame(() => {
            const scrollPercent = window.scrollY / (document.body.scrollHeight - window.innerHeight);
            updateReadingProgress(scrollPercent);
            isScrolling = false;
        });
        isScrolling = true;
    }
});
```

## Practical Examples

Real-world patterns for interactive websites:

```js
// Article reading progress indicator
function initReadingProgress() {
    const progressBar = document.querySelector('.reading-progress');
    const article = document.querySelector('article');
    
    if (!progressBar || !article) return;
    
    function updateProgress() {
        const articleTop = article.offsetTop;
        const articleHeight = article.offsetHeight;
        const scrollTop = window.scrollY;
        const windowHeight = window.innerHeight;
        
        const progress = Math.max(0, Math.min(1, 
            (scrollTop - articleTop + windowHeight) / articleHeight
        ));
        
        progressBar.style.transform = `scaleX(${progress})`;
    }
    
    window.addEventListener('scroll', updateProgress);
    updateProgress(); // Initial call
}

// Newsletter signup with validation
function initNewsletterForm() {
    const form = document.querySelector('.newsletter-form');
    const emailInput = form.querySelector('input[type="email"]');
    const submitButton = form.querySelector('button[type="submit"]');
    
    form.addEventListener('submit', async (event) => {
        event.preventDefault();
        
        const email = emailInput.value.trim();
        if (!isValidEmail(email)) {
            showFormError('Please enter a valid email address');
            return;
        }
        
        submitButton.disabled = true;
        submitButton.textContent = 'Subscribing...';
        
        try {
            await subscribeToNewsletter(email);
            showFormSuccess('Thank you! You\'re now subscribed.');
            form.reset();
        } catch (error) {
            showFormError('Subscription failed. Please try again.');
        } finally {
            submitButton.disabled = false;
            submitButton.textContent = 'Subscribe';
        }
    });
}

function isValidEmail(email) {
    return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
}

// Initialize when page loads
document.addEventListener('DOMContentLoaded', () => {
    initReadingProgress();
    initNewsletterForm();
});
```

## Best Practices

Write maintainable JavaScript:

1. **Use meaningful names**
   ```js
   // Good
   const calculateTotalPrice = (items) => { /* ... */ };
   
   // Bad
   const calc = (x) => { /* ... */ };
   ```

2. **Keep functions small and focused**
   ```js
   // Good - single responsibility
   function validateEmail(email) {
       return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(email);
   }
   
   function showErrorMessage(message) {
       const errorDiv = document.querySelector('.error-message');
       errorDiv.textContent = message;
       errorDiv.style.display = 'block';
   }
   ```

3. **Use const by default, let when needed**
4. **Handle errors gracefully**
5. **Use modern features like async/await**
6. **Comment complex logic**
7. **Test your code in different browsers**
